║ Query 3: Sales Today                                                  ║
║ WHERE: created_at = TODAY                                             ║
║ AGGREGATION: COUNT(*), SUM(quantity)                                  ║
║ PERFORMANCE: O(n) - Single table scan                                 ║
║                                                                        ║
║ Query 4: Low Stock Products                                           ║
║ WHERE: current_stock <= minimum_stock                                 ║
║ AGGREGATION: COUNT + ORDER BY deficit                                 ║
║ PERFORMANCE: O(n) - WHERE + ORDER                                     ║
║                                                                        ║
║ Query 5: Expiring Batches (7 days)                                    ║
║ LEFT JOIN: stock_out_transactions                                     ║
║ WHERE: expiry_date BETWEEN today AND today+7                          ║
║ AGGREGATION: GROUP BY, SUM, ORDER BY date                             ║
║ PERFORMANCE: O(n log n) - Optimal with indexes                        ║
║                                                                        ║
║ Query 6: Batch Traceability                                           ║
║ WHERE: id = {batch_id} OR batch_number = {number}                     ║
║ WITH: product relationship, stock_out_transactions                    ║
║ PERFORMANCE: O(1) - Indexed PRIMARY KEY lookup                        ║
║                                                                        ║
